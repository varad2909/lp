#include <iostream>
#include <queue>
#include <iomanip>
#include <algorithm>
using namespace std;

struct Process {
    int id;
    int arrivalTime;
    int burstTime;
    int remainingTime;
    int completionTime;
    int turnAroundTime;
    int waitingTime;
    int startTime;
};

int main() {
    int n, tq;
    cout << "Enter the number of processes: ";
    cin >> n;

    cout << "Enter the time quantum: ";
    cin >> tq;

    Process p[50];
    for (int i = 0; i < n; i++) {
        p[i].id = i + 1;
        cout << "Enter arrival time of Process " << i + 1 << ": ";
        cin >> p[i].arrivalTime;
        cout << "Enter burst time of Process " << i + 1 << ": ";
        cin >> p[i].burstTime;
        p[i].remainingTime = p[i].burstTime;
        p[i].completionTime = 0;
        p[i].waitingTime = 0;
        p[i].turnAroundTime = 0;
        p[i].startTime = -1;
    }

    // Sort by arrival time
    sort(p, p + n, [](Process a, Process b) { return a.arrivalTime < b.arrivalTime; });

    queue<int> q;
    vector<int> mark(n, 0);
    int currentTime = 0, completed = 0;
    float totalWT = 0, totalTAT = 0;

    q.push(0);
    mark[0] = 1;

    cout << "\n===== Round Robin Scheduling Simulation =====\n";

    while (completed < n) {
        if (q.empty()) {
            for (int i = 0; i < n; i++) {
                if (p[i].remainingTime > 0) {
                    q.push(i);
                    mark[i] = 1;
                    break;
                }
            }
        }

        int idx = q.front();
        q.pop();

        // Set start time if first execution
        if (p[idx].startTime == -1)
            p[idx].startTime = max(currentTime, p[idx].arrivalTime);

        // CPU idle till process arrives
        currentTime = max(currentTime, p[idx].arrivalTime);

        // Execute for time quantum or remaining burst
        if (p[idx].remainingTime > tq) {
            p[idx].remainingTime -= tq;
            currentTime += tq;
        } else {
            currentTime += p[idx].remainingTime;
            p[idx].remainingTime = 0;
            p[idx].completionTime = currentTime;
            p[idx].turnAroundTime = p[idx].completionTime - p[idx].arrivalTime;
            p[idx].waitingTime = p[idx].turnAroundTime - p[idx].burstTime;
            totalWT += p[idx].waitingTime;
            totalTAT += p[idx].turnAroundTime;
            completed++;
        }

        // Add newly arrived processes to queue
        for (int i = 0; i < n; i++) {
            if (!mark[i] && p[i].arrivalTime <= currentTime && p[i].remainingTime > 0) {
                q.push(i);
                mark[i] = 1;
            }
        }

        // Reinsert the process if not completed
        if (p[idx].remainingTime > 0)
            q.push(idx);
    }

    // Display table
    cout << "\n==============================================\n";
    cout << "PID\tAT\tBT\tCT\tTAT\tWT\n";
    cout << "==============================================\n";
    for (int i = 0; i < n; i++) {
        cout << "P" << p[i].id << "\t"
             << p[i].arrivalTime << "\t"
             << p[i].burstTime << "\t"
             << p[i].completionTime << "\t"
             << p[i].turnAroundTime << "\t"
             << p[i].waitingTime << endl;
    }
    cout << "==============================================\n";
    cout << fixed << setprecision(2);
    cout << "Average Turnaround Time: " << (totalTAT / n) << endl;
    cout << "Average Waiting Time:    " << (totalWT / n) << endl;

    return 0;
}
