#include <iostream>
#include <iomanip>
using namespace std;

int main() {
    int n;

    cout << "Enter the number of processes: ";
    cin >> n;

    int at[20], bt[20], ct[20], tat[20], wt[20];
    float avg_tat = 0, avg_wt = 0;

    cout << "Enter Arrival Time for " << n << " processes:\n";
    for (int i = 0; i < n; i++) {
        cout << "P" << i + 1 << ": ";
        cin >> at[i];
    }

    cout << "Enter Burst Time for " << n << " processes:\n";
    for (int i = 0; i < n; i++) {
        cout << "P" << i + 1 << ": ";
        cin >> bt[i];
    }

    // Sort according to Arrival Time (FCFS)
    for (int i = 0; i < n - 1; i++) {
        for (int j = i + 1; j < n; j++) {
            if (at[i] > at[j]) {
                swap(at[i], at[j]);
                swap(bt[i], bt[j]);
            }
        }
    }

    // Calculate Completion Time (CT)
    ct[0] = at[0] + bt[0];
    for (int i = 1; i < n; i++) {
        if (at[i] > ct[i - 1])
            ct[i] = at[i] + bt[i];
        else
            ct[i] = ct[i - 1] + bt[i];
    }

    // Calculate Turnaround Time (TAT) and Waiting Time (WT)
    for (int i = 0; i < n; i++) {
        tat[i] = ct[i] - at[i];
        wt[i] = tat[i] - bt[i];
        avg_tat += tat[i];
        avg_wt += wt[i];
    }

    // Display Table
    cout << "\n==============================================\n";
    cout << "Process\tAT\tBT\tCT\tTAT\tWT\n";
    cout << "==============================================\n";
    for (int i = 0; i < n; i++) {
        cout << "P" << i + 1 << "\t"
             << at[i] << "\t"
             << bt[i] << "\t"
             << ct[i] << "\t"
             << tat[i] << "\t"
             << wt[i] << endl;
    }

    cout << "==============================================\n";
    cout << fixed;
    cout.precision(2);
    cout << "Average Turnaround Time: " << (avg_tat / n) << endl;
    cout << "Average Waiting Time:    " << (avg_wt / n) << endl;

    return 0;
}

P3: 8
